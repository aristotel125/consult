// Расчет колеса баланса с 10-балльной шкалой

function calculateWheel() {

  // Матрица комбинаций для расчета баллов (масштабируем до 7)

  const scoreMatrix = {

    'YY': 7, 'YU': 5, 'UY': 5, 'YN': 4,

    'NY': 4, 'UU': 3, 'UN': 3, 'NU': 3, 'NN': 2

  };

  

  // Сферы и соответствующие вопросы

  const spheres = {

    'Работа/Цели': [1, 2],

    'Финансы': [3, 4],

    'Самореализация': [5, 6],

    'Отношения': [7, 8],

    'Баланс/Энергия': [9, 10]

  };

  

  // Рассчитываем только если еще не было расчета

  if (Object.keys(wheelData).length === 0) {

    for (let sphere in spheres) {

      const questions = spheres[sphere];

      const a1 = answers[questions[0]] || 'U';

      const a2 = answers[questions[1]] || 'U';

      const combo = a1 + a2;

      // Округляем до целого числа

      wheelData[sphere] = Math.round(scoreMatrix[combo] || 3);

    }

  }

  

  // Создаем ползунки для управления

  const slidersContainer = document.getElementById('wheelSliders');

  slidersContainer.innerHTML = '';

  

  for (let sphere in wheelData) {

    const value = wheelData[sphere];

    

    const sliderGroup = document.createElement('div');

    sliderGroup.className = 'slider-group';

    sliderGroup.innerHTML = `

      <span class="slider-label">${sphere}</span>

      <input type="range" class="slider" min="0" max="10" step="1" 

             value="${value}" 

             oninput="updateWheelValue('${sphere}', this.value)">

      <span class="slider-value">${value}/10</span>

    `;

    slidersContainer.appendChild(sliderGroup);

  }

  

  // Рисуем колесо

  drawWheel();

}


// Обновление значения колеса

  sliderGroups.forEach(group => {

    if (group.querySelector('.slider-label').textContent === sphere) {

      const valueSpan = group.querySelector('.slider-value');

// Рисование колеса баланса

function drawWheel() {

  const canvas = document.getElementById('wheelCanvas');

  const ctx = canvas.getContext('2d');

  const centerX = canvas.width / 2;

  const centerY = canvas.height / 2;

  const radius = 150;

  

  ctx.clearRect(0, 0, canvas.width, canvas.height);

  

  // Рисуем сетку для 10 уровней

  ctx.strokeStyle = '#E2E8F0';

  ctx.lineWidth = 1;

  

  for (let r = 1; r <= 10; r++) {

    ctx.beginPath();

    ctx.arc(centerX, centerY, (radius / 10) * r, 0, 2 * Math.PI);

    ctx.stroke();

  }

  

  // Рисуем оси

  const labels = Object.keys(wheelData);

  const angleStep = (2 * Math.PI) / labels.length;

  

  labels.forEach((label, i) => {

    const angle = -Math.PI / 2 + angleStep * i;

    const x = centerX + Math.cos(angle) * radius;

    const y = centerY + Math.sin(angle) * radius;

    

    ctx.beginPath();

    ctx.moveTo(centerX, centerY);

    ctx.lineTo(x, y);

    ctx.stroke();

    

    // Подписи

    ctx.font = '13px sans-serif';

    ctx.fillStyle = '#0F172A';

    ctx.textAlign = 'center';

    ctx.textBaseline = 'middle';

    

    const labelX = centerX + Math.cos(angle) * (radius + 30);

    const labelY = centerY + Math.sin(angle) * (radius + 30);

    

    // Разбиваем длинные надписи

    const words = label.split('/');

    if (words.length > 1) {

      ctx.fillText(words[0], labelX, labelY - 7);

      ctx.fillText(words[1], labelX, labelY + 7);

    } else {

      ctx.fillText(label, labelX, labelY);

    }

  });

  

  // Рисуем основную фигуру колеса

  ctx.beginPath();

  

  labels.forEach((label, i) => {

    const angle = -Math.PI / 2 + angleStep * i;

    const value = wheelData[label];

    const r = (radius / 10) * value;

    const x = centerX + Math.cos(angle) * r;

    const y = centerY + Math.sin(angle) * r;

    

    if (i === 0) {

      ctx.moveTo(x, y);

    } else {

      ctx.lineTo(x, y);

    }

  });

  

  ctx.closePath();

  

  // Заливка и обводка - всегда синяя

  ctx.fillStyle = 'rgba(37, 99, 235, 0.3)';

  ctx.strokeStyle = '#2563EB';

  ctx.lineWidth = 2;

  ctx.fill();

  ctx.stroke();

  

  // Рисуем точки на вершинах - все синие

  labels.forEach((label, i) => {

    const angle = -Math.PI / 2 + angleStep * i;

    const value = wheelData[label];

    const r = (radius / 10) * value;

    const x = centerX + Math.cos(angle) * r;

    const y = centerY + Math.sin(angle) * r;

    

    ctx.beginPath();

    ctx.arc(x, y, 4, 0, 2 * Math.PI);

    ctx.fillStyle = '#2563EB';

    ctx.fill();

  });

}


// Расчет ценностей с 5-балльной шкалой и округлением

function calculateValues() {

  // Получаем баллы сфер из колеса баланса

  const scoreMatrix = {

    'YY': 7, 'YU': 5, 'UY': 5, 'YN': 4,

    'NY': 4, 'UU': 3, 'UN': 3, 'NU': 3, 'NN': 2

  };

  

  // Расчет баллов сфер

  const sphereScores = {

    work: scoreMatrix[(answers[1] || 'U') + (answers[2] || 'U')] || 3,

    finance: scoreMatrix[(answers[3] || 'U') + (answers[4] || 'U')] || 3,

    realization: scoreMatrix[(answers[5] || 'U') + (answers[6] || 'U')] || 3,

    relations: scoreMatrix[(answers[7] || 'U') + (answers[8] || 'U')] || 3,

    balance: scoreMatrix[(answers[9] || 'U') + (answers[10] || 'U')] || 3

  };

  

  // Расчет ценностей на 5-балльной шкале с округлением до 0.5

  if (Object.keys(valuesData).length === 0) {

    const roundToHalf = (num) => Math.round(num * 2) / 2;

    

    valuesData = {

      'Смысл': roundToHalf((sphereScores.realization / 7) * 5),

      'Польза': roundToHalf((sphereScores.relations / 7) * 5),

      'Деньги': roundToHalf((sphereScores.finance / 7) * 5),

      'Свобода': roundToHalf(((sphereScores.balance + sphereScores.work) / 14) * 5),

      'Система': roundToHalf(((sphereScores.work + sphereScores.realization) / 14) * 5)

    };

  }

  

  // Создаем вертикальный эквалайзер

  createValuesEqualizer();

}


// Создание улучшенного вертикального эквалайзера

function createValuesEqualizer() {

  const container = document.getElementById('valuesEqualizer');

  // Сохраняем метки шкалы

  const scaleMarks = container.querySelector('.scale-marks');

  container.innerHTML = '';

  if (scaleMarks) {

    container.appendChild(scaleMarks);

  }

  

  const fullNames = {

    'Смысл': 'Смысл и предназначение',

    'Польза': 'Польза людям',

    'Деньги': 'Деньги без предательства',

    'Свобода': 'Автономия и гибкость',

    'Система': 'Системность и результат'

  };

  

  for (let valueName in valuesData) {

    const column = document.createElement('div');

    column.className = 'eq-column';

    

    const value = valuesData[valueName];

    const percentage = (value / 5) * 100;

    

    column.innerHTML = `

      <div class="eq-bar-container">

        <div class="eq-bar-bg"></div>

        <div class="eq-bar" style="height: ${percentage}%" data-value="${valueName}">

          <span class="eq-value">${value}</span>

        </div>

      </div>

      <div class="eq-controls">

        <button class="eq-btn" onclick="changeValue('${valueName}', -0.5)">−</button>

        <button class="eq-btn" onclick="changeValue('${valueName}', 0.5)">+</button>

      </div>

      <div class="eq-label">${fullNames[valueName]}</div>

    `;

    

    container.appendChild(column);

  }

}


// Изменение значения ценности кнопками

function changeValue(valueName, delta) {

  let newValue = valuesData[valueName] + delta;

  newValue = Math.max(0, Math.min(5, newValue));

  newValue = Math.round(newValue * 2) / 2; // Округление до 0.5

  

  valuesData[valueName] = newValue;

  updateEqualizer();

}


// Обновление визуализации эквалайзера

function updateEqualizer() {

  for (let valueName in valuesData) {

    const value = valuesData[valueName];

    const percentage = (value / 5) * 100;

    

    // Находим элементы

    const bar = document.querySelector(`.eq-bar[data-value="${valueName}"]`);

    const valueSpan = bar.querySelector('.eq-value');

    

    // Обновляем

    bar.style.height = percentage + '%';

    valueSpan.textContent = value;

  }

}

function calcIncomeNew(){

  const levelSelect = document.getElementById("level");

  const daysPerWeekInput = document.getElementById("daysPerWeek");

  const consultationsInput = document.getElementById("consultations");

  

  if (!levelSelect || !daysPerWeekInput || !consultationsInput) return;

  

  const rate = parseFloat(levelSelect.value) || 3000;

  const daysPerWeek = parseInt(daysPerWeekInput.value) || 5;

  const consultations = parseInt(consultationsInput.value) || 4;


  const daysPerMonth = daysPerWeek * 4; // считаем месяц как 4 недели

  const daysPerMonthSpan = document.getElementById("daysPerMonth");

  if (daysPerMonthSpan) {

    daysPerMonthSpan.innerText = daysPerMonth;

  }


  const base = daysPerMonth * consultations * rate;

  const activeSkills = document.querySelectorAll(".skill.active").length;

  const final = base * (1 + activeSkills * 0.1);

  

  const finalIncomeElement = document.getElementById("finalIncomeCalc");

  if (finalIncomeElement) {

    finalIncomeElement.innerText = Math.round(final).toLocaleString('ru-RU');

  }

}
